<div id="box">
    <p> {{ count }} </p>

        <button @click="handleClick()">add</button>
        <button @click="handleClick">add</button> //不写小括号 传不了对象 ，自动传进当前的事件对象
        <button @click="count = count + 1">add</button>

        <ul @click.self = "clickUl">      //点击ul自身才会调用，点击li不会调用
            <li @click.stop="clickLi( )"></li> //阻止冒泡
             <li @click.capture="clickLi( )"></li> //事件捕获
        </ul>

        <a @click.prevent = "" > </a>   //阻止默认行为

        <a @click.once = "" > </a>   //只能触发一次 (付款 )

        <input @keyup = "handleKey()">  
        <imput @keyup.enter = "">    //只按下enter才调用
        <imput @keyup.97 = "">    //只按下keycode==97  (a)才调用

//同时按才触发
</div>
