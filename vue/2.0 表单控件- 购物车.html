v-model 双向数据绑定  MVVM
vue 是单向数据流，v-model本质上不过是语法糖。



<div id="box">
    <input tupe="checkbox" v-model="isAllCheck" @change="change">
    <ul>
        <li v-for="data in datalist">
            <input type="checkbox" v-model="checkgroup" :value="data"/>
            {{data}}
        </li>
    </ul>
    <p>{{sum}}</p>
</div>
<script>
    var vm = new Vue({
        el:"#box",
        data:{
            name:"xiaoming",
            checkgroup:[],
            datalist:[
                {
                    name:"1",
                    price:200,
                    num:1
                },
                {
                    name:"2",
                    price:400,
                    num:6
                },
                {
                    name:"3",
                    price:500,
                    num:1
                }

            ]
            
        },
        computed:{
            sum(){
                var summary = 0;
                for(var i in this.checkgroup){
                    summary+= this.checkgroup[i].num*this.checkgroup[i].price
                }
                return summary;
            }
        },
        methods:{
            change(){
                if(isAllChecked){
                    vm.checkgroup = this.datalist
                }else{
                    vm.checkgroup = []
                }
            }
        }
        
    })
</script>
