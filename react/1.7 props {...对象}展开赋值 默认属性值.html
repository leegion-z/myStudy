1.7 props {...对象}展开赋值 /默认属性值


属性和状态是react中数据传递的载体

属性：
props = properties
属性：一个事物的性质与关系
属性往往是与生俱来、无法自己改变的

（1）属性是声明后不允许修改的东西
（2）属性只能在组件初始化时候声明并传入组件的内部，并在组件内部通过this.props获取
（3）组件内部可以通过getDefaultProps声明默认属性

        /*
            (1)在父组件上通过 key=“value” 写属性

            (2)子 通过this.props获取属性
            这样组件的可复用性提高了。

            (3) 注意在传参数时候，如果写成isShow="true" 那么这是一个字符串
                如果写成isShow={true} 这个是布尔值

            (4){...对象}展开赋值

            (5)默认属性值
            *.defaultProps = {
    
            }

            (6)
            import propTypes from "prop-types";
            *.propTypes={
                name : propTypes.string,
                age:propTypes.number
            }
        */


import ReactDOM from "react-dom";
import React,{Component} from "react";

class Child extends Components{
    render( ){
        return <div>
            child
            { this.props.name }   //子 通过this.props获取属性
            { this.props.age }

            {
                this.props.show ?
                <div>显示</div>
                : null
            }
        </div>
    }
}

class Hello extends Component{
    render ( ) {
        return <div>
            123
            <Child name="legion" age="100" /> //在父组件上通过 key=“value” 写属性
            <Child name="lee" age="100"  show={true}/> //不能使用show="true" 会当成字符串 ，要用 show={false}
        </div>
    }
}
ReactDOM.render(<Hello/>,document.getElementById("box"));

import ReactDOM from "react-dom";
import React,{Component} from "react";
import  propTypes  form " prop-types ";   //自动下载了

class Child extends Components{
    render(){
        return <div>child

            {
                this.props.show ?
                <div>显示</div>
                : null
            }

        </div>
    }
}

Child.defaultProps = {     
    show : true
}

Child.propTypes = {
    show : propTypes . bool  //规定属性show  必须是boolean 类型  否则会报错
}

class Hello extends Component{
    render ( ) {
        return <div>
            123

            <Child name="lee" age="100"  show="false"/>  //会发出警告 
        </div>
    }
}
ReactDOM.render(<Hello/>,document.getElementById("box"));
