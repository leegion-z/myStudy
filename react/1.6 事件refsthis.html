call\apply\bind 回顾：

          第一个参数  第二个参数（可以不写）
fn.call ( 新的指向，  原函数实参，   原函数实参，   原函数实参...)  //立即执行，第二个参数往后是实参，可以写多个
fn.apply( 新的指向，  一个实参/原函数实参数组)                   //立即执行，多个参数时第二个参数是实参集合[] ，可以写多个子项，适合操作不定参
fn.bind( 新的指向 )( 实参 ，实参 ...)                              //不立即执行，第二个（）可以不写，但写了第二个（）才调用执行
或者：
fn.bind( 新的指向，  原函数实参，   原函数实参，   原函数实参... )()


import ReactDOM from "react-dom";
import React,{Component} from "react";


class Hello extends Component{
    render(){
        return （
                <input type="text" ref="username"/>
                
                <button onMouseOver={ this.legion }>click</button>  //绑定事件不能执行 ，例：{this.legion( )}
                <button onClick= { this.getusername.bind(this , "xxx" , "yyy" ) }>click</button>
            ）  //this指向丢了，要用.bind指向组件
    }
    
    legion(){}   //要调用的函数，写在render的同一级
    getusername(){
        console.log(this.refs.username.value);
    }
}
ReactDOM.render(<Hello/>,document.getElementById("box"));


//改变指向的第二种方式:constructor （传参不方便）
class Hello extends Component{

    constructor(){   //构造器 初始化函数 可以继承父类的所有属性 只会执行一次 但不容易传参
        super();
        this.getusername = this.getusername.bind(this);  //先修正this指向  注意：= 要赋值，覆盖原函数
    }

    render(){
        return （
                <input type="text" ref="username"/>

                <button onClick= { this.getusername }>click</button>
            ）  
    }
    

    getusername(){
        console.log(this.refs.username.value);
    }
}

ReactDOM.render(<Hello/>,document.getElementById("box"));


//改变指向的第三种方式：箭头函数

            } }>click</button>

            ）  
    }
    
}

ReactDOM.render(<Hello/>,document.getElementById("box"));
