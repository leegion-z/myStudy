请求别的服务器：
///////////////////////////////////////////////get
var https =require("https");
const options = {
  hostname: 'www.lagou.com';;,
  port: 443,
  path: '/',
  method: 'get',
  // headers: {
  //   'Content-Type': 'application/x-www-form-urlencoded',
  //   'Content-Length': Buffer.byteLength(postData)
  // }
};
var req  = https.request(options,(res)=>{
    var html = "";
    res.on("data",function(chunk){
      html+= chunk;
    })
    res.on("end",function(){
        console.log(html);
    })
})
req.end();

var http =require("http");
var postData  ="do=digg&action=1&id=898";
const options = {
  hostname: 'd.onlinedown.net',
  port: 80,
  path: '/ajax_top_1.4.php',
  method: 'post',
  headers: {
    'Content-Type': 'application/x-www-form-urlencoded',
    'Content-Length': Buffer.byteLength(postData)
  }
};
var req  = http.request(options,(res)=>{
    var html = "";
    res.on("data",function(chunk){
      html+= chunk;
    })
    res.on("end",function(){
        console.log(html);
    })
})
req.write(postData);
req.end();
