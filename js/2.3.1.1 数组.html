    function every( elements , fn ){
        var resultBool = [];
        for(var i=0;i<elements.length;i++){
            resultBool.push(fn( elements[i] ));    
        }
        for(var i=0;i<resultBool.length;i++){
            if(resultBool[i] == false){
                return false;
            }
        }
        return true;
    }
