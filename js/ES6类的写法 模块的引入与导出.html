class person{
    constructor(name,age){
        this.name = name;
        this.age = age;
    }

    say(){
        console.log(this.name);
    }
}

var obj = new person("legion",18);

es6:  extends
class Chinese extends person{
    constructor(name,age,color){
        super(name,age);
        this.color = color;
    }
    say(){
        //同名覆盖，不同名方法继承
    }
}
var obj = new chinese("xiaoming",10,"yellow");
