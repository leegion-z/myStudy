1、基本语法：
<?php
    //注释和JS类似

2、输出到浏览器：
<?php
    echo 'hello';
    echo '<br>';

    print 'hello';
    print_r();
?>
echo \ print \ print_r( ) 区别：
    echo 可以输出多个变量；
    print只能输出一个；
    print_r( )可以输出详细的信息，数组和对象
<?php
    echo 'hello','world';
    print 'hello';

    $arr = array(1,2,3);
    print_r($arr);
?>

3、变量：
    $xxx： $ 表示要使用一个变量
<?php
    $foo = 'hello';
?>

4、作用域：
    $GLOBALS    global
<?php
    $x = 10;
    function foo(){
        $x = 20;
        echo $x;   //输出20，与JS不同：如果内部不定义，外面的相同变量名也找不到
        echo $GLOBALS['x'];   //这样直接输出10
        global $x;  //全局 把外部变量引进内部调用，如果重新赋值相同名的变量$x='', 就会被修改
        echo $x     //输出10
    }
    foo();
?>

5、数据类型：

    （1）字符串：

    $foo = 'world';
    $bar = 'hello' . $foo;  //链接用点. 
    $bac = "hello $foo !!!"  //" "双引号 相连可以不加点，识别空格

    //索引 查找字符串
    echo $foo[2]; //输出r

    //字符串长度 strlen（）
    echo strlen($foo);  //输出5

    //位置 strpos（ ， ）
    echo strpos($foo,'r'); //输出2

    （2）数值：

    $foo = 123;
    $aoo = 1.2;
    $show = true;

    //判断类型 var_dump()
    echo var_dump($foo); //输出 int 123
    echo var_dump($aoo); //输出 float 1.2
    echo var_dump($show); //输出 boolean true


    （3）数组：
            a、数值数组
            b、关联数组

    //数值数组
    $arr = array('a','b','c');

    //通过索引 查找
    echo $arr[1]; //输出b

    //数组长度
    echo count($arr); //输出3

    //添加数值
    $arr[] = 'd';  //[]内不写，添加到最后
    $arr[6] = 'e'; //添加索引值为6 的是 e， 可以替换

    //关联数组
    $arr = array (
        'name'=>'hello',
        'age'=>'20'
    );

    //数组长度
    ehco count($arr); //输出3

    //查找
    ehco $arr['name'];    //输出'hello'

    //添加内容
    $arr['job'] = 'it';


    （4）对象：
类 类似构造函数
    class Foo{
        function show(){
            return 1;
        }
    }

    $foo = new Foo();
    echo $foo->show(); //输出1  js中foo.show
